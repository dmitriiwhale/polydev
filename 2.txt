async getWalletReport(): Promise<void> { 
    try { 
        const allOrders = await this.getAllOrders();
        const closedPositions = allOrders.filter(order => 
            order.status === 'closed' && order.realizedPnl !== null
        );
        
        const totalTrades = closedPositions.length;
        const winningTrades = closedPositions.filter(p => p.realizedPnl > 0).length;
        const winRate = totalTrades > 0 ? (winningTrades / totalTrades) * 100 : 0;
        
        const totalPnl = closedPositions.reduce((sum, p) => sum + p.realizedPnl, 0);
        const avgProfitPerWin = winningTrades > 0 
            ? closedPositions.filter(p => p.realizedPnl > 0)
                .reduce((sum, p) => sum + p.realizedPnl, 0) / winningTrades
            : 0;
        
        const tradingProfile = this.analyzeTradingBehavior(closedPositions);
        
        const report = {
            wallet: this.walletAddress,
            period: 'all-time',
            metrics: {
                totalTrades,
                winningTrades,
                winRate: parseFloat(winRate.toFixed(2)),
                totalPnl: parseFloat(totalPnl.toFixed(4)),
                avgProfitPerWin: parseFloat(avgProfitPerWin.toFixed(4)),
                bestTrade: Math.max(...closedPositions.map(p => p.realizedPnl)),
                worstTrade: Math.min(...closedPositions.map(p => p.realizedPnl)),
                sharpeRatio: this.calculateSharpeRatio(closedPositions),
                maxDrawdown: this.calculateMaxDrawdown(closedPositions),
                volatility: this.calculateVolatility(closedPositions)
            },
            behavior: tradingProfile,
            generatedAt: new Date().toISOString(),
            chain: 'polygon',
            protocol: 'polymarket'
        };
        
        await this.persistReport(report);
        this.emit('wallet-report:generated', report);
        
        console.log(`‚úÖ Wallet Report Generated: ${this.walletAddress}`);
        console.log(`   üìà ${report.metrics.winRate}% Win Rate | ${report.metrics.totalTrades} trades`);
        console.log(`   üí∞ P&L: $${report.metrics.totalPnl} | Sharpe: ${report.metrics.sharpeRatio}`);
        console.log(`   üéØ Profile: ${report.behavior.profileType}`);
        
        return;
        
    } catch (error) { 
        const errorContext = {
            method: 'getWalletReport',
            wallet: this.walletAddress,
            timestamp: new Date().toISOString(),
            error: error.message,
            stackTrace: error.stack
        };
        
        console.error('‚ùå Wallet Report Generation Failed:', errorContext);
        await this.sendErrorNotification(errorContext);
        
        throw new Error(`Wallet report generation failed: ${error.message}`);
    }
}
